# RA2 Modder Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local Flask web app for inspecting and editing all RA2/YR (Mental Omega 3.3) game object definitions, with sprite previews and INI patch export.

**Architecture:** MIX archives extracted via `ra2mix` at startup, rules/art INI chain merged and indexed into SQLite (~2–5 sec). Sprites rendered on-demand per request (numpy + Pillow) and cached to disk. Flask + Jinja2 + HTMX serves a dark-themed card-grid browser + tabbed property editor. Edits write to `mypatch/rulesmd.ini` without touching original MIX files.

**Tech Stack:** Python 3.11, Poetry, Flask 3, ra2mix 1.0, Pillow 10, numpy 1.26, SQLite (stdlib), HTMX 1.x (CDN)

```python
def parse_ini(text: str) -> dict[str, dict[str, str]]:
    """Parse RA2 INI text.

    Handles: duplicate keys (last wins), duplicate sections (merged),
    keys without values, inline ; comments.
    """
    result: dict[str, dict[str, str]] = {}
    current: str | None = None

    for raw in text.splitlines():
        line = raw.strip()
        if not line or line.startswith(";"):
            continue

        # Strip inline comment
        ci = line.find(";")
        if ci > 0:
            line = line[:ci].strip()

        if line.startswith("[") and line.endswith("]"):
            current = line[1:-1].strip()
            if current not in result:
                result[current] = {}
        elif current is not None:
            if "=" in line:
                key, _, value = line.partition("=")
                result[current][key.strip()] = value.strip()
            else:
                result[current][line] = ""

    return result
```

```python
import os
import tempfile
from pathlib import Path

import ra2mix

from ra2modder.config import GameConfig

# Load order: base game first, Mental Omega expansions in number order
_MIX_LOAD_ORDER = [
    "ra2.mix",
    "ra2md.mix",
    *(f"expandmo{n:02d}.mix" for n in range(94, 100)),
]


def load_game_files(config: GameConfig) -> dict[str, bytes]:
    """Extract all game files from MIX archives.

    Later files in load order override earlier ones.
    Nested .mix files found inside archives are resolved recursively.
    """
    result: dict[str, bytes] = {}

    for mix_name in _MIX_LOAD_ORDER:
        mix_path = config.game_dir / mix_name
        if not mix_path.exists():
            continue
        try:
            files = ra2mix.read(str(mix_path))
        except Exception:
            continue

        result.update(files)

        # Resolve any nested .mix archives
        for name, data in list(files.items()):
            if name.lower().endswith(".mix"):
                nested = _read_nested_mix(data)
                result.update(nested)

    return result


def _read_nested_mix(data: bytes) -> dict[str, bytes]:
    """Write bytes to a temp file, extract via ra2mix, delete temp file."""
    fd, tmp = tempfile.mkstemp(suffix=".mix")
    try:
        os.write(fd, data)
        os.close(fd)
        return ra2mix.read(tmp)
    except Exception:
        return {}
    finally:
        try:
            os.unlink(tmp)
        except OSError:
            pass
```

```python
from pathlib import Path
from ra2modder.ini.parser import parse_ini

_BASE_NAMES = ["rules.ini", "rulesmd.ini"]
_EXPAND_NAMES = [f"rulesmo{n:02d}.ini" for n in range(94, 100)]


def load_rules(
    game_files: dict[str, bytes], patch_dir: Path
) -> dict[str, dict[str, str]]:
    """Merge rules INI chain: base → MO expansions → user patch."""
    merged: dict[str, dict[str, str]] = {}
    lower = {k.lower(): v for k, v in game_files.items()}

    for name in _BASE_NAMES + _EXPAND_NAMES:
        data = lower.get(name)
        if data:
            _merge(merged, parse_ini(data.decode("latin-1", errors="replace")))

    patch = patch_dir / "rulesmd.ini"
    if patch.exists():
        _merge(merged, parse_ini(patch.read_text(encoding="latin-1")))

    return merged


def _merge(base: dict, override: dict) -> None:
    for section, keys in override.items():
        base.setdefault(section, {}).update(keys)
```

```python
import struct
from ra2modder.csf.reader import parse_csf


def _make_csf(labels: dict[str, str]) -> bytes:
    """Build a minimal valid CSF binary."""
    # Header: magic(4) + version(4) + n_labels(4) + n_strings(4) + unused(4) + lang(4)
    n = len(labels)
    header = b"CSF " + struct.pack("<5I", 3, n, n, 0, 0)

    body = b""
    for name, text in labels.items():
        name_b = name.encode("ascii")
        # Label block: " LBL" + n_strings(4) + name_len(4) + name
        body += b" LBL" + struct.pack("<II", 1, len(name_b)) + name_b
        # String block: " RTS" + value_len(4) + encoded_utf16le_xor
        encoded = b"".join(
            struct.pack("<H", ord(ch) ^ 0xFFFF) for ch in text
        )
        body += b" RTS" + struct.pack("<I", len(text)) + encoded

    return header + body
```

```python
import struct


def parse_csf(data: bytes) -> dict[str, str]:
    """Parse a .csf string table file into {LABEL_KEY: display_string}."""
    if len(data) < 24 or data[:4] != b"CSF ":
        return {}

    _version, n_labels, _n_strings, _unused, _lang = struct.unpack_from(
        "<5I", data, 4
    )

    pos = 24
    result: dict[str, str] = {}

    for _ in range(n_labels):
        if pos + 12 > len(data) or data[pos:pos+4] != b" LBL":
            break
        pos += 4

        n_strings = struct.unpack_from("<I", data, pos)[0]; pos += 4
        name_len = struct.unpack_from("<I", data, pos)[0]; pos += 4
        name = data[pos:pos+name_len].decode("ascii", errors="replace").upper()
        pos += name_len

        for _ in range(n_strings):
            if pos + 8 > len(data):
                break
            tag = data[pos:pos+4]; pos += 4
            val_len = struct.unpack_from("<I", data, pos)[0]; pos += 4

            if tag in (b" RTS", b"WRTS"):
                raw = data[pos:pos + val_len * 2]
                pos += val_len * 2
                words = struct.unpack_from(f"<{val_len}H", raw)
                decoded = "".join(chr(w ^ 0xFFFF) for w in words)
                result[name] = decoded

                if tag == b"WRTS":
                    extra = struct.unpack_from("<I", data, pos)[0]; pos += 4
                    pos += extra
            else:
                pos += val_len * 2

    return result
```

```python
import json
import sqlite3

# Maps rules.ini list-section name → object type label
OBJECT_TYPE_MAP = {
    "VehicleTypes": "VehicleType",
    "InfantryTypes": "InfantryType",
    "AircraftTypes": "AircraftType",
    "BuildingTypes": "BuildingType",
    "WeaponTypes": "WeaponType",
    "Warheads": "Warhead",
    "Projectiles": "Projectile",
    "SuperWeaponTypes": "SuperWeaponType",
    "Countries": "Country",
    "Houses": "House",
}

_SIDE_KEYS = {"Soviets": "Soviet", "Americans": "Allied", "British": "Allied",
              "French": "Allied", "Germans": "Allied", "Koreans": "Allied",
              "YuriCountry": "Yuri", "Arabs": "Yuri", "Africans": "Yuri",
              "Russians": "Soviet", "Confederation": "Soviet", "Foehn": "Foehn"}


def build_index(
    rules: dict[str, dict[str, str]],
    art: dict[str, dict[str, str]],
    strings: dict[str, str],
) -> sqlite3.Connection:
    conn = sqlite3.connect(":memory:")
    conn.row_factory = sqlite3.Row
    conn.execute("""
        CREATE TABLE objects (
            id    TEXT PRIMARY KEY,
            type  TEXT NOT NULL,
            name  TEXT,
            side  TEXT,
            props TEXT NOT NULL,
            art   TEXT NOT NULL
        )
    """)
    conn.execute("""
        CREATE TABLE patch_changes (
            id        INTEGER PRIMARY KEY AUTOINCREMENT,
            object_id TEXT NOT NULL,
            key       TEXT NOT NULL,
            old_value TEXT,
            new_value TEXT,
            timestamp TEXT NOT NULL
        )
    """)
    conn.execute("CREATE INDEX idx_type ON objects(type)")
    conn.execute("CREATE INDEX idx_name ON objects(name)")

    for list_section, obj_type in OBJECT_TYPE_MAP.items():
        id_map = rules.get(list_section, {})
        for obj_id in id_map.values():
            if not obj_id:
                continue
            props = rules.get(obj_id, {})
            art_props = art.get(obj_id, {})
            display_name = (
                strings.get(f"NAME:{obj_id}")
                or strings.get(f"NOSTR:{obj_id}")
                or props.get("Name", obj_id)
            )
            owner = props.get("Owner", "")
            side = _SIDE_KEYS.get(owner, "")
            conn.execute(
                "INSERT OR REPLACE INTO objects VALUES (?,?,?,?,?,?)",
                (obj_id, obj_type, display_name, side,
                 json.dumps(props), json.dumps(art_props)),
            )

    conn.commit()
    return conn
```

```python
import json
import sqlite3


def _row_to_dict(row: sqlite3.Row) -> dict:
    d = dict(row)
    d["props"] = json.loads(d["props"])
    d["art"] = json.loads(d["art"])
    return d


def list_objects(conn: sqlite3.Connection, obj_type: str) -> list[dict]:
    rows = conn.execute(
        "SELECT * FROM objects WHERE type=? ORDER BY name", (obj_type,)
    ).fetchall()
    return [_row_to_dict(r) for r in rows]


def get_object(conn: sqlite3.Connection, obj_id: str) -> dict | None:
    row = conn.execute(
        "SELECT * FROM objects WHERE id=?", (obj_id,)
    ).fetchone()
    return _row_to_dict(row) if row else None


def search_objects(
    conn: sqlite3.Connection,
    query: str = "",
    obj_type: str = "",
    side: str = "",
) -> list[dict]:
    sql = "SELECT * FROM objects WHERE 1=1"
    params: list = []
    if query:
        sql += " AND (LOWER(id) LIKE ? OR LOWER(name) LIKE ?)"
        q = f"%{query.lower()}%"
        params += [q, q]
    if obj_type:
        sql += " AND type=?"
        params.append(obj_type)
    if side:
        sql += " AND side=?"
        params.append(side)
    sql += " ORDER BY name"
    return [_row_to_dict(r) for r in conn.execute(sql, params).fetchall()]
```

```python
@objects_bp.route("/<obj_type>")
def object_list(obj_type: str):
    if obj_type not in _TYPE_SLUGS:
        abort(404)
    db = current_app.config["DB"]
    obj_type_label = _TYPE_SLUGS[obj_type]
    query = request.args.get("q", "")
    side = request.args.get("side", "")
    if query or side:
        objects = search_objects(db, query=query, obj_type=obj_type_label, side=side)
    else:
        objects = list_objects(db, obj_type_label)

    if request.headers.get("HX-Request"):
        return render_template("partials/unit_cards.html",
                               objects=objects, obj_type=obj_type)
    return render_template("pages/object_list.html",
                           objects=objects, obj_type=obj_type,
                           obj_type_label=obj_type_label,
                           type_slugs=_TYPE_SLUGS)


@objects_bp.route("/<obj_type>/<obj_id>", methods=["GET", "POST"])
def object_detail(obj_type: str, obj_id: str):
    if obj_type not in _TYPE_SLUGS:
        abort(404)
    db = current_app.config["DB"]
    obj = get_object(db, obj_id)
    if obj is None:
        abort(404)

    if request.method == "POST":
        from ra2modder.patch.manager import save_field
        config = current_app.config["GAME_CONFIG"]
        for key, new_val in request.form.items():
            old_val = obj["props"].get(key, "")
            if new_val != old_val:
                save_field(config.patch_dir, obj_id, key, old_val, new_val)
        # Re-fetch after save
        obj = get_object(db, obj_id)

    return render_template("partials/detail_tabs.html",
                           obj=obj, obj_type=obj_type)
```

```python
@sprites_bp.route("/<obj_id>/cameo.png")
def cameo(obj_id: str):
    config = current_app.config["GAME_CONFIG"]
    cached = get_cached(config.cache_dir, obj_id, "cameo")
    if cached:
        return send_file(io.BytesIO(cached), mimetype="image/png")

    game_files = current_app.config["GAME_FILES"]
    from ra2modder.db.queries import get_object
    obj = get_object(current_app.config["DB"], obj_id)
    if obj is None:
        abort(404)

    art = obj.get("art", {})
    cameo_name = (art.get("Cameo") or obj_id + "ICON").upper() + ".SHP"
    shp_data = {k.lower(): v for k, v in game_files.items()}.get(cameo_name.lower())
    if not shp_data:
        abort(404)

    from ra2modder.render.shp import render_shp
    from ra2modder.render.palette import load_palette
    pal_data = {k.lower(): v for k, v in game_files.items()}.get("cameo.pal", b"")
    palette = load_palette(pal_data) if pal_data else _grey_palette()
    img = render_shp(shp_data, palette)
    buf = io.BytesIO()
    img.save(buf, format="PNG")
    png = buf.getvalue()
    save_cached(config.cache_dir, obj_id, "cameo", png)
    return send_file(io.BytesIO(png), mimetype="image/png")
```

```python
def load_palette(data: bytes) -> list[tuple[int, int, int]]:
    """Load a 768-byte .pal file into a list of 256 (R, G, B) 8-bit tuples.

    RA2 palettes store 6-bit values (0-63); multiply by 4 to get 0-255.
    Index 0 is always transparent (black).
    """
    if len(data) < 768:
        # Fallback: greyscale
        return [(i, i, i) for i in range(256)]

    palette = []
    for i in range(256):
        r = data[i * 3] * 4
        g = data[i * 3 + 1] * 4
        b = data[i * 3 + 2] * 4
        palette.append((r, g, b))

    # Index 0 = transparent (conventionally black)
    palette[0] = (0, 0, 0)
    return palette
```

```python
import struct
from PIL import Image


def render_shp(
    data: bytes,
    palette: list[tuple[int, int, int]],
    frame_index: int = 0,
) -> Image.Image:
    """Render a SHP TD frame to a PIL RGBA Image.

    Supports uncompressed frames (high bit of offset = 0).
    Format80-compressed frames (high bit = 1) are decompressed first.
    Index 0 is transparent.
    """
    try:
        return _render(data, palette, frame_index)
    except Exception:
        return Image.new("RGBA", (48, 48), (0, 0, 0, 0))


def _render(
    data: bytes,
    palette: list[tuple[int, int, int]],
    frame_index: int,
) -> Image.Image:
    if len(data) < 12:
        return Image.new("RGBA", (48, 48), (0, 0, 0, 0))

    num_frames, _, _, width, height, _ = struct.unpack_from("<6H", data, 0)
    if num_frames == 0 or frame_index >= num_frames:
        return Image.new("RGBA", (width or 48, height or 48), (0, 0, 0, 0))

    offset_table_start = 12
    raw_offset = struct.unpack_from("<I", data, offset_table_start + frame_index * 4)[0]

    compressed = bool(raw_offset & 0x80000000)
    file_offset = raw_offset & 0x7FFFFFFF
    frame_size = width * height

    if file_offset + frame_size > len(data) and not compressed:
        return Image.new("RGBA", (width, height), (0, 0, 0, 0))

    if compressed:
        frame_bytes = _decompress_format80(data[file_offset:], frame_size)
    else:
        frame_bytes = data[file_offset: file_offset + frame_size]

    # Map palette indices → RGBA pixels
    pixels = []
    for idx in frame_bytes[:frame_size]:
        if idx == 0:
            pixels.append((0, 0, 0, 0))
        else:
            r, g, b = palette[idx % 256]
            pixels.append((r, g, b, 255))

    img = Image.new("RGBA", (width, height))
    img.putdata(pixels)
    return img


def _decompress_format80(src: bytes, expected_size: int) -> bytes:
    """Format 80 (LCW) decompression used in RA2/TS SHP files."""
    dst: list[int] = []
    sp = 0

    while sp < len(src) and len(dst) < expected_size:
        cmd = src[sp]; sp += 1

        if cmd == 0x80:
            break

        if (cmd & 0x80) == 0:
            # Short relative copy: 12-bit offset, 3-10 byte count
            count = ((cmd >> 4) & 0x07) + 3
            lo = src[sp]; sp += 1
            off = ((cmd & 0x0F) << 8) | lo
            start = len(dst) - off
            for i in range(count):
                dst.append(dst[start + i] if 0 <= start + i < len(dst) else 0)

        elif (cmd & 0xC0) == 0x80:
            # Raw copy: cmd & 0x3F bytes follow directly
            count = cmd & 0x3F
            if count == 0:
                break
            dst.extend(src[sp: sp + count])
            sp += count

        elif cmd == 0xFE:
            # Fill run
            count = src[sp] | (src[sp + 1] << 8); sp += 2
            color = src[sp]; sp += 1
            dst.extend([color] * count)

        elif cmd == 0xFF:
            # Large relative copy
            count = src[sp] | (src[sp + 1] << 8); sp += 2
            off = src[sp] | (src[sp + 1] << 8); sp += 2
            start = len(dst) - off
            for i in range(count):
                dst.append(dst[start + i] if 0 <= start + i < len(dst) else 0)

        else:
            # Medium absolute copy: position in output
            count = (cmd & 0x3F) + 3
            pos = src[sp] | (src[sp + 1] << 8); sp += 2
            for i in range(count):
                dst.append(dst[pos + i] if 0 <= pos + i < len(dst) else 0)

    return bytes(dst[:expected_size])
```

```python
import struct
import numpy as np
from PIL import Image

_VXL_MAGIC = b"Voxel Animation\x00"
_CANVAS = 128
# RA2 isometric: SE view, ~30° elevation
_ANGLE = 4 * np.pi / 8   # 135° from north = SE
_ELEV = 30.0 * np.pi / 180.0


def render_vxl(
    vxl_data: bytes,
    hva_data: bytes | None,
    palette: list[tuple[int, int, int]],
    facing: int = 4,
) -> Image.Image:
    """Render a VXL voxel model to a PIL RGBA Image.

    facing: 0–7 (0=N, clockwise). Default 4 = SE (standard RA2 game view).
    """
    try:
        return _render(vxl_data, hva_data, palette, facing)
    except Exception:
        return Image.new("RGBA", (_CANVAS, _CANVAS), (0, 0, 0, 0))


def _render(vxl_data, hva_data, palette, facing):
    sections = _parse_vxl(vxl_data)
    if not sections:
        return Image.new("RGBA", (_CANVAS, _CANVAS), (0, 0, 0, 0))

    canvas = np.zeros((_CANVAS, _CANVAS, 4), dtype=np.uint8)
    depth = np.full((_CANVAS, _CANVAS), np.inf)

    angle = facing * np.pi / 4
    cos_a, sin_a = np.cos(angle), np.sin(angle)
    cos_e, sin_e = np.cos(_ELEV), np.sin(_ELEV)

    for sec in sections:
        scale = sec["scale"] if sec["scale"] > 0 else 1.0
        cx = sec["x_size"] / 2.0
        cy = sec["y_size"] / 2.0
        cz = sec["z_size"] / 2.0

        for (vx, vy, vz, cidx, _nidx) in sec["voxels"]:
            # Centre the voxels around origin
            wx = (vx - cx) * scale
            wy = (vy - cy) * scale
            wz = (vz - cz) * scale

            # Isometric projection
            sx = cos_a * wx - sin_a * wy
            sy = sin_a * sin_e * wx + cos_a * sin_e * wy + cos_e * wz
            sz = cos_a * wx - sin_a * wy  # depth

            px = int(sx + _CANVAS // 2)
            py = int(-sy + _CANVAS // 2)

            if 0 <= px < _CANVAS and 0 <= py < _CANVAS:
                if sz < depth[py, px]:
                    depth[py, px] = sz
                    r, g, b = palette[cidx % 256]
                    canvas[py, px] = [r, g, b, 255]

    return Image.fromarray(canvas, "RGBA")


def _parse_vxl(data: bytes) -> list[dict]:
    if len(data) < 32 or not data[:16].startswith(b"Voxel Animation"):
        return []

    n_sections = struct.unpack_from("<I", data, 20)[0]
    body_size = struct.unpack_from("<I", data, 28)[0]
    pos = 32

    # Palette (768 bytes, embedded — we ignore it and use caller's palette)
    pos += 768

    header_start = pos
    pos += n_sections * 28   # section headers

    tailer_start = pos
    _TAILER = 92
    pos += n_sections * _TAILER

    body_start = pos

    sections = []
    for i in range(n_sections):
        t = tailer_start + i * _TAILER
        ss_off = struct.unpack_from("<I", data, t)[0]
        se_off = struct.unpack_from("<I", data, t + 4)[0]
        vd_off = struct.unpack_from("<I", data, t + 8)[0]
        scale = struct.unpack_from("<f", data, t + 84)[0]
        x_size, y_size, z_size = struct.unpack_from("<BBB", data, t + 88)

        col_count = x_size * y_size
        ss_base = body_start + ss_off
        se_base = body_start + se_off
        vd_base = body_start + vd_off

        if ss_base + col_count * 4 > len(data):
            continue

        span_starts = struct.unpack_from(f"<{col_count}i", data, ss_base)
        span_ends = struct.unpack_from(f"<{col_count}i", data, se_base)

        voxels = []
        for col in range(col_count):
            vx = col % x_size
            vy = col // x_size
            s_off = span_starts[col]
            e_off = span_ends[col]
            if s_off < 0 or e_off <= s_off:
                continue

            vd_pos = vd_base + s_off
            vd_end = vd_base + e_off

            while vd_pos < vd_end and vd_pos + 1 < len(data):
                skip = data[vd_pos]; vd_pos += 1
                count = data[vd_pos]; vd_pos += 1
                if count == 0:
                    break
                vz = skip
                for _ in range(count):
                    if vd_pos + 1 >= len(data):
                        break
                    color_idx = data[vd_pos]; vd_pos += 1
                    normal_idx = data[vd_pos]; vd_pos += 1
                    voxels.append((vx, vy, vz, color_idx, normal_idx))
                    vz += 1
                vd_pos += 1  # closing count byte

        sections.append({
            "x_size": x_size, "y_size": y_size, "z_size": z_size,
            "scale": scale, "voxels": voxels,
        })

    return sections
```

```python
from pathlib import Path


def cache_path(cache_dir: Path, obj_id: str, kind: str) -> Path:
    """Return the cache file path for a rendered sprite."""
    cache_dir.mkdir(parents=True, exist_ok=True)
    return cache_dir / f"{obj_id}_{kind}.png"


def get_cached(cache_dir: Path, obj_id: str, kind: str) -> bytes | None:
    """Return cached PNG bytes if available, else None."""
    p = cache_path(cache_dir, obj_id, kind)
    if p.exists():
        return p.read_bytes()
    return None


def save_cached(cache_dir: Path, obj_id: str, kind: str, png: bytes) -> None:
    """Save rendered PNG bytes to disk cache."""
    cache_path(cache_dir, obj_id, kind).write_bytes(png)


def invalidate_cached(cache_dir: Path, obj_id: str) -> None:
    """Delete all cached sprites for an object (e.g. after palette change)."""
    for p in cache_dir.glob(f"{obj_id}_*.png"):
        p.unlink(missing_ok=True)
```

```python
from ra2modder.ini.parser import parse_ini


def serialise_patch(sections: dict[str, dict[str, str]]) -> str:
    """Convert {section: {key: value}} to INI text."""
    lines = []
    for section, keys in sorted(sections.items()):
        if not keys:
            continue
        lines.append(f"[{section}]")
        for k, v in sorted(keys.items()):
            lines.append(f"{k}={v}")
        lines.append("")
    return "\n".join(lines)


def read_patch(patch_path) -> dict[str, dict[str, str]]:
    """Read existing patch INI or return empty dict."""
    from pathlib import Path
    p = Path(patch_path)
    if not p.exists():
        return {}
    return parse_ini(p.read_text(encoding="latin-1"))
```

```python
from pathlib import Path
from ra2modder.patch.writer import serialise_patch, read_patch


def save_field(
    patch_dir: Path,
    obj_id: str,
    key: str,
    old_value: str,
    new_value: str,
) -> None:
    """Write a single changed field to the patch INI."""
    patch_dir.mkdir(parents=True, exist_ok=True)
    patch_file = patch_dir / "rulesmd.ini"
    sections = read_patch(patch_file)
    sections.setdefault(obj_id, {})[key] = new_value
    patch_file.write_text(serialise_patch(sections), encoding="latin-1")

    # Track old value in a sidecar for diffing
    _write_original(patch_dir, obj_id, key, old_value)


def revert_field(patch_dir: Path, obj_id: str, key: str) -> None:
    """Remove a single field from the patch INI."""
    patch_file = patch_dir / "rulesmd.ini"
    sections = read_patch(patch_file)
    if obj_id in sections:
        sections[obj_id].pop(key, None)
        if not sections[obj_id]:
            del sections[obj_id]
    patch_file.write_text(serialise_patch(sections), encoding="latin-1")

    originals_file = patch_dir / ".originals.ini"
    originals = read_patch(originals_file)
    if obj_id in originals:
        originals[obj_id].pop(key, None)
    originals_file.write_text(serialise_patch(originals), encoding="latin-1")


def get_diff(
    patch_dir: Path,
) -> dict[str, dict[str, tuple[str, str]]]:
    """Return {section: {key: (old_value, new_value)}} for all patch changes."""
    patch_file = patch_dir / "rulesmd.ini"
    if not patch_file.exists():
        return {}

    patch = read_patch(patch_file)
    originals = read_patch(patch_dir / ".originals.ini")

    result: dict[str, dict[str, tuple[str, str]]] = {}
    for section, keys in patch.items():
        for key, new_val in keys.items():
            old_val = originals.get(section, {}).get(key, "")
            result.setdefault(section, {})[key] = (old_val, new_val)
    return result


def _write_original(
    patch_dir: Path, obj_id: str, key: str, old_value: str
) -> None:
    """Store original value for diffing (only if not already stored)."""
    originals_file = patch_dir / ".originals.ini"
    originals = read_patch(originals_file)
    section = originals.setdefault(obj_id, {})
    if key not in section:
        section[key] = old_value
    originals_file.write_text(serialise_patch(originals), encoding="latin-1")
```

```python
assets_bp = Blueprint("assets", __name__, url_prefix="/assets")


@assets_bp.route("/")
def asset_browser():
    game_files = current_app.config["GAME_FILES"]
    by_ext: dict[str, list[str]] = {}
    for name in sorted(game_files.keys()):
        ext = name.rsplit(".", 1)[-1].upper() if "." in name else "OTHER"
        by_ext.setdefault(ext, []).append(name)
    return render_template("pages/assets.html", by_ext=by_ext)


@assets_bp.route("/preview")
def preview():
    name = request.args.get("file", "").lower()
    game_files = current_app.config["GAME_FILES"]
    lower = {k.lower(): v for k, v in game_files.items()}
    data = lower.get(name)
    if data is None:
        return "<p style='color:var(--red)'>File not found.</p>"

    if name.endswith(".ini") or name.endswith(".txt"):
        text = data.decode("latin-1", errors="replace")[:4000]
        return f"<pre style='font-size:10px;max-height:400px;overflow:auto'>{text}</pre>"

    return (
        f"<p style='color:var(--muted)'>{len(data):,} bytes — "
        f"binary preview not available for this type.</p>"
    )
```
